CREATE TABLE users(
	uid SERIAL NOT NULL,
	email varchar(30) UNIQUE NOT NULL,
	name varchar(20) NOT NULL,
	password varchar(30) NOT NULL,
	rid int,
	FOREIGN KEY(rid) REFERENCES role(rid) ON DELETE CASCADE,
	PRIMARY KEY(uid)
);
